using Microsoft.UI.Xaml.Input;
using Microsoft.UI.Xaml.Shapes;
using System.Collections.ObjectModel;
using Windows.Foundation;

namespace McTimeline;

public record TimelineItem(string KeyId, string Title, string Description, DateTime Start, DateTime End);

public class McTimeline : Control
{
    public ObservableCollection<TimelineItem> Items { get; } = new();

    public DateTime MinTime
    {
        get => (DateTime)GetValue(MinTimeProperty);
        set => SetValue(MinTimeProperty, value);
    }
    public static readonly DependencyProperty MinTimeProperty =
        DependencyProperty.Register(nameof(MinTime), typeof(DateTime), typeof(McTimeline), new PropertyMetadata(DateTime.MinValue));

    public DateTime MaxTime
    {
        get => (DateTime)GetValue(MaxTimeProperty);
        set => SetValue(MaxTimeProperty, value);
    }
    public static readonly DependencyProperty MaxTimeProperty =
        DependencyProperty.Register(nameof(MaxTime), typeof(DateTime), typeof(McTimeline), new PropertyMetadata(DateTime.MaxValue));

    public double PixelsPerHour
    {
        get => (double)GetValue(PixelsPerHourProperty);
        set => SetValue(PixelsPerHourProperty, value);
    }
    public static readonly DependencyProperty PixelsPerHourProperty =
        DependencyProperty.Register(nameof(PixelsPerHour), typeof(double), typeof(McTimeline), new PropertyMetadata(50d));

    public double RowHeight { get; set; } = 40;

    private ScrollViewer? _timelineScroll;
    private Canvas? _canvas;
    private Canvas? _minimapCanvas;
    private Border? _minimapViewport;
    private Grid? _minimapRoot;

    private readonly Dictionary<string, int> _seriesOrder = new();
    private double _minimapScale;
    private bool _isMiniDrag;
    private Point _lastMiniPointer;

    public McTimeline()
    {
        DefaultStyleKey = typeof(McTimeline);
    }

    public ICommand ScrollToItemCommand { get; private set; } = null!;
    public ICommand ScrollToTimeCommand { get; private set; } = null!;
    public ICommand EnsureVisibleCommand { get; private set; } = null!;

    protected override void OnApplyTemplate()
    {
        base.OnApplyTemplate();

        _timelineScroll = GetTemplateChild("PART_TimelineScroll") as ScrollViewer;
        _canvas = GetTemplateChild("PART_TimelineCanvas") as Canvas;
        _minimapCanvas = GetTemplateChild("PART_MinimapCanvas") as Canvas;
        _minimapViewport = GetTemplateChild("PART_MinimapViewport") as Border;
        _minimapRoot = GetTemplateChild("PART_MinimapRoot") as Grid;

        if (_minimapViewport != null)
        {
            _minimapViewport.PointerPressed += MinimapViewport_PointerPressed;
            _minimapViewport.PointerMoved += MinimapViewport_PointerMoved;
            _minimapViewport.PointerReleased += MinimapViewport_PointerReleased;
        }

        ScrollToItemCommand = new RelayCommand<TimelineItem>(i => ScrollToItem(i, true));
        ScrollToTimeCommand = new RelayCommand<DateTime>(t => ScrollToTime(t, true));
        EnsureVisibleCommand = new RelayCommand<TimelineItem>(EnsureItemVisible);

        RenderTimeline();
        RenderMinimap();
    }

    private void RenderTimeline()
    {
        if (_canvas == null)
            return;
        _canvas.Children.Clear();

        _seriesOrder.Clear();
        int row = 0;
        foreach (var series in Items.GroupBy(i => i.KeyId))
        {
            _seriesOrder[series.Key] = row;
            foreach (var item in series)
            {
                double x = GetXForItem(item);
                double w = GetWidthForItem(item);
                double y = row * RowHeight;

                var rect = new Rectangle
                {
                    Width = Math.Max(w, 1),
                    Height = RowHeight - 4,
                    Fill = new SolidColorBrush(Windows.UI.Colors.SteelBlue),
                    Stroke = new SolidColorBrush(Windows.UI.Colors.Black),
                    StrokeThickness = 1
                };
                Canvas.SetLeft(rect, x);
                Canvas.SetTop(rect, y);
                _canvas.Children.Add(rect);
            }
            row++;
        }

        _canvas.Width = (MaxTime - MinTime).TotalHours * PixelsPerHour;
        _canvas.Height = row * RowHeight;
    }

    private void RenderMinimap()
    {
        if (_minimapCanvas == null)
            return;
        _minimapCanvas.Children.Clear();

        double totalHours = (MaxTime - MinTime).TotalHours;
        double miniWidth = _minimapCanvas.ActualWidth;
        if (miniWidth <= 0)
            miniWidth = ActualWidth;
        double miniHeight = _minimapCanvas.ActualHeight;
        _minimapScale = miniWidth / (totalHours * PixelsPerHour);

        double rowHeight = miniHeight / Math.Max(1, _seriesOrder.Count);

        foreach (var kvp in _seriesOrder)
        {
            int row = kvp.Value;
            foreach (var item in Items.Where(i => i.KeyId == kvp.Key))
            {
                double x = GetXForItem(item) * _minimapScale;
                double w = GetWidthForItem(item) * _minimapScale;
                var rect = new Rectangle
                {
                    Width = Math.Max(w, 1),
                    Height = Math.Max(rowHeight - 2, 1),
                    Fill = new SolidColorBrush(Windows.UI.Colors.SteelBlue),
                    Opacity = 0.6
                };
                Canvas.SetLeft(rect, x);
                Canvas.SetTop(rect, row * rowHeight);
                _minimapCanvas.Children.Add(rect);
            }
        }

        UpdateMinimapViewport();
    }

    private void UpdateMinimapViewport()
    {
        if (_timelineScroll == null || _minimapViewport == null)
            return;
        double miniLeft = _timelineScroll.HorizontalOffset * _minimapScale;
        double miniWidth = _timelineScroll.ViewportWidth * _minimapScale;
        _minimapViewport.Width = miniWidth;
        Canvas.SetLeft(_minimapViewport, miniLeft);
    }

    private void MinimapViewport_PointerPressed(object sender, PointerRoutedEventArgs e)
    {
        _isMiniDrag = true;
        _lastMiniPointer = e.GetCurrentPoint(_minimapCanvas).Position;
        _minimapViewport!.CapturePointer(e.Pointer);
    }
    private void MinimapViewport_PointerMoved(object sender, PointerRoutedEventArgs e)
    {
        if (!_isMiniDrag || _timelineScroll == null)
            return;
        var pos = e.GetCurrentPoint(_minimapCanvas).Position;
        double dx = pos.X - _lastMiniPointer.X;
        _lastMiniPointer = pos;
        double newOffset = _timelineScroll.HorizontalOffset + dx / _minimapScale;
        _timelineScroll.ChangeView(newOffset, null, null, true);
        UpdateMinimapViewport();
    }
    private void MinimapViewport_PointerReleased(object sender, PointerRoutedEventArgs e)
    {
        _isMiniDrag = false;
        _minimapViewport!.ReleasePointerCapture(e.Pointer);
    }

    public void ScrollToItem(TimelineItem item, bool center = false)
    {
        double x = GetXForItem(item);
        double w = GetWidthForItem(item);
        double targetX = center ? x + (w / 2) - (_timelineScroll!.ViewportWidth / 2) : x;
        double targetY = GetYForItem(item);
        SmoothScrollTo(targetX, targetY);
        UpdateMinimapViewport();
    }
    public void ScrollToItem(string keyId, bool center = false)
    {
        var item = Items.FirstOrDefault(i => i.KeyId == keyId);
        if (item != null)
            ScrollToItem(item, center);
    }
    public void ScrollToTime(DateTime time, bool center = false)
    {
        double hours = (time - MinTime).TotalHours;
        double x = hours * PixelsPerHour;
        if (center)
            x -= (_timelineScroll!.ViewportWidth / 2);
        SmoothScrollTo(x, null);
        UpdateMinimapViewport();
    }
